19static char *add_prefix_dir(const char *path, const char *dir)
20{
21 char *path_prefixed = NULL;
22 size_t len_path_file = 0;
23
24 path_prefixed = strdup(path);
25 if (!path_prefixed)
26 return NULL;
27 len_path_file = strlen(path);
28 if (path_prefixed[0] == '.' && path_prefixed[1] == '/') {
29 memmove(path_prefixed, path_prefixed + 2, len_path_file - 1);
30 realloc_and_concat_before(dir, &path_prefixed);
31 realloc_and_concat_before("./", &path_prefixed);
32 }
33 else {
34 realloc_and_concat_before(dir, &path_prefixed);
35 }
36 return path_prefixed;
37}

84static int get_hash_and_path_with_extension_if_exists(
