21// We can safely don't verify the return value of all the functions
22// because all functions don't raise errors if they receives bad parameters
23// and for the "homemade" free_functions, we don't care if the pointer is NULLs
24// This functions doesn't respect the 25 lines rule because it's a little bit
25// complicated to split it in multiple functions. It's not a good practice to
26// split a function in multiple functions if it's not possible to do it without
27// losing the readability of the code
28list_t *mygit_merge_or_get_conflicts(const char *remote_branch,
29 const char *message)
30{
31 work_tree_t *wt_curr_br = NULL;
32 work_tree_t *wt_remote_br = NULL;
33 work_tree_t *wt_merged = NULL;
34 char *hash_wt_merged = NULL;
35 char *curr_branch = NULL;
36 list_t *conflicted_files = NULL;
37 commit_t *commit_merge = NULL;
38 char *ref_curr_branch = NULL;
39 char *ref_remote_branch = NULL;
40 char *hash_commit = NULL;
41
42 if (!remote_branch)
43 return NULL;
44 // Get the current branch and init the conflicted files list
45 curr_branch = get_current_branch();
46 wt_curr_br = get_work_tree_from_branch(curr_branch);
47 wt_remote_br = get_work_tree_from_branch(remote_branch);
48 conflicted_files = create_init_list();
49 wt_merged =
50 mygit_merge_work_trees(wt_curr_br, wt_remote_br, &conflicted_files);
51 // If there is conflict, we returns the list of conflicted files and free
52 // memory
53 if (*conflicted_files) {
54 free_work_tree(wt_curr_br);
55 free_work_tree(wt_remote_br);
56 free_work_tree(wt_merged);
57 free(curr_branch);
58 return conflicted_files;
59 }
60 // If there is no conflict, we can save the work tree and create the merge
61 // commit
62 hash_wt_merged = create_blob_of_work_tree(wt_merged);
63 commit_merge = create_commit_with_tree_key_val(hash_wt_merged);
64 // We update the keys of the commit to match with the merge commit
65 ref_remote_branch = get_ref(remote_branch);
66 ref_curr_branch = get_ref(curr_branch);
67 if (message)
68 insert_key_val_in_commit(commit_merge, COMMIT_KEY_MESSAGE, message);
69 else
70 insert_key_val_in_commit(commit_merge, COMMIT_KEY_MESSAGE, "Merge commit");
71 insert_key_val_in_commit(commit_merge, COMMIT_KEY_MERGED_PARENT,
72 ref_remote_branch);
73 insert_key_val_in_commit(commit_merge, COMMIT_KEY_PARENT, ref_curr_branch);
74 // We create the blob of the commit and update the refs
75 hash_commit = create_blob_of_commit(commit_merge);
76 create_or_update_ref(curr_branch, hash_commit);
77 create_or_update_ref(MYGIT_FILE_REF_HEAD, hash_commit);
78 // We delete the merged branch and restore the work tree of the current branch
79 delete_ref(remote_branch);
80 restore_work_tree(wt_merged, PREFIX_PATH);
81 // We free memory and return NULL because there is no conflict
82 free_work_tree(wt_curr_br);
83 free_work_tree(wt_remote_br);
84 free_work_tree(wt_merged);
85 free_commit(commit_merge);
86 free(hash_wt_merged);
87 free(hash_commit);
88 free(ref_curr_branch);
89 free(ref_remote_branch);
90 free(curr_branch);
91 free(conflicted_files);
92 return NULL;
93}
