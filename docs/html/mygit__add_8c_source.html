38int mygit_add(const char *file)
39{
40 work_tree_t *work_tree = NULL;
41 int append_wt_ret = 0;
42
43 if (!file || !does_file_exists(file))
44 return 0;
45 work_tree = get_add_work_tree_or_init();
46 if (!work_tree)
47 return 0;
48 append_wt_ret = append_work_tree(work_tree, file, NULL, 0);
49 if (append_wt_ret) {
50 work_tree_to_file(work_tree, MYGIT_PATH_ADD);
51 free_work_tree(work_tree);
52 return 1;
53 }
54 if (append_wt_ret == 0)
55 fprintf(stderr, "Error: file '%s' already added\n", file);
56 else
57 fprintf(stderr, "Error: could not add file '%s'\n", file);
58 free_work_tree(work_tree);
59 return 0;
60}
