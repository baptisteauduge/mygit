57static char *fill_new_commit_create_blob_and_get_hash(commit_t *new_commit,

74char *mygit_commit(const char *branch_name, const char *message)
75{
76 int pre_checks = 0;
77 commit_t *new_commit = NULL;
78 char *hash_new_commit = NULL;
79
80 if (!branch_name)
81 return NULL;
82 pre_checks = check_if_refs_exists() && check_if_branch_exists(branch_name) &&
83 check_if_head_is_in_branch(branch_name);
84 if (!pre_checks)
85 return NULL;
86 new_commit = get_commit_from_added();
87 if (!new_commit)
88 return NULL;
89 hash_new_commit = fill_new_commit_create_blob_and_get_hash(
90 new_commit, message, branch_name);
91 free_commit(new_commit);
92 if (!hash_new_commit)
93 return NULL;
94 set_head_and_branch_hash_last_commit(branch_name, hash_new_commit);
95 return hash_new_commit;
96}

98static int get_commits_from_ref_and_append_list(list_t **list_all_commits,
99 const char *ref)
100{
101 list_t *list_current_commits = NULL;
102 list_t *tmp = NULL;
103
104 list_current_commits = get_commit_from_branch_list(ref);
105 if (!list_current_commits)
106 return 0;
107 tmp = concat_list_without_duplicate(*list_all_commits, list_current_commits);
108 if (!tmp) {
109 free_list(list_current_commits);
110 return 0;
111 }
112 free_list(*list_all_commits);
113 free_list(list_current_commits);
114 *list_all_commits = tmp;
115 return 1;
116}

118list_t *get_list_all_commits(void)
119{
120 list_t *list_all_refs = get_list_files_and_dir(MYGIT_DIR_REFS);
121 cell_t *current_ref = NULL;
122 list_t *list_all_commits = NULL;
123
124 if (!list_all_refs)
125 return NULL;
126 list_all_commits = create_init_list();
127 if (!list_all_commits) {
128 free_list(list_all_refs);
129 return NULL;
130 }
131 current_ref = *list_all_refs;
132 while (current_ref) {
133 if (!get_commits_from_ref_and_append_list(&list_all_commits,
134 current_ref->data)) {
135 free_list(list_all_refs);
136 free_list(list_all_commits);
137 return NULL;
138 }
139 current_ref = current_ref->next;
140 }
141 free_list(list_all_refs);
142 return list_all_commits;
143}
