44static void mygit_choose_for_each_file(const list_t *conflicts,
45 const char *message, const char *branch)
46{
47 list_t *conflicts_delete_current_br = NULL;
48 list_t *conflicts_delete_other_br = NULL;
49 cell_t *current_cell = NULL;
50
51 if (!conflicts || !branch)
52 return;
53 conflicts_delete_current_br = create_init_list();
54 conflicts_delete_other_br = create_init_list();
55 if (!conflicts_delete_current_br || !conflicts_delete_other_br)
56 return;
57 current_cell = *conflicts;
58 while (current_cell) {
59 LOG_INFO(
60 "Do you want to keep the current branch changes for '%s' ? (y/n)\n",
61 current_cell->data);
62 if (mygit_get_yes_no_answer()) {
63 create_and_insert_cell_in_list(conflicts_delete_other_br,
64 current_cell->data);
65 }
66 else {
67 create_and_insert_cell_in_list(conflicts_delete_current_br,
68 current_cell->data);
69 }
70 current_cell = current_cell->next;
71 }
72 mygit_create_deletion_commit(branch, conflicts_delete_other_br, message);
73 mygit_keep_other_branch_changes(conflicts_delete_current_br, message);
74 free_list(conflicts_delete_current_br);
75 free_list(conflicts_delete_other_br);
76}

78static void mygit_handle_answer(const char *branch, const list_t *conflicts,
79 const char *message)
80{
81 int answer = 0;
82
83 while (answer < 1 || answer > 3) {
84 scanf("%d", &answer);
85 if (answer < 1 || answer > 3)
86 LOG_ERROR("Invalid answer. Please choose between 1 and 3.\n");
87 }
88 if (answer == 1) {
89 mygit_create_deletion_commit(branch, conflicts, message);
90 }
91 else if (answer == 2) {
92 mygit_keep_other_branch_changes(conflicts, message);
93 }
94 else {
95 mygit_choose_for_each_file(conflicts, message, branch);
96 }
97 mygit_merge(branch, message);
98}

113void mygit_merge(const char *branch, const char *message)
114{
115 list_t *conflicts = NULL;
116
117 if (!branch || !does_branch_exists(branch)) {
118 LOG_ERROR("The branch '%s' does not exists.\n", branch);
119 return;
120 }
121 conflicts = mygit_merge_or_get_conflicts(branch, message);
122 if (conflicts) {
123 LOG_ERROR("There is conflicts between the branches that you are trying to "
124 "merge.\n");
125 mygit_handle_conflicts(branch, conflicts, message);
126 free_list(conflicts);
127 }
128 else
129 LOG_INFO("Merge done.\n");
130}
