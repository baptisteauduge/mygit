36void mygit_create_deletion_commit(const char *branch, const list_t *conflicts,
37 const char *message)
38{
39 commit_t *last_commit_branch = NULL;
40 work_tree_t *last_work_tree_branch = NULL;
41 char *default_branch = NULL;
42
43 if (!branch || !conflicts)
44 return;
45 default_branch = get_current_branch();
46 if (!default_branch)
47 return;
48 if (!mygit_checkout_branch(branch))
49 return;
50 last_commit_branch = get_last_commit_from_branch(branch);
51 last_work_tree_branch = get_work_tree_from_commit(last_commit_branch);
52 remove(MYGIT_PATH_ADD);
53 add_files_not_in_conflict(conflicts, last_work_tree_branch);
54 if (!message)
55 free(mygit_commit(branch, "mygit: deletion commit"));
56 else
57 free(mygit_commit(branch, message));
58 mygit_checkout_branch(default_branch);
59 free(default_branch);
60 free_commit(last_commit_branch);
61 free_work_tree(last_work_tree_branch);
62}
