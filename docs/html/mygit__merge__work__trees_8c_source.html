14static int is_in_conflict(work_tree_t *wt, work_file_t *wf)
15{
16 if (!wt || !wf)
17 return 0;
18 for (size_t i = 0; i < wt->size; ++i) {
19 if (strcmp(wt->tab[i].name, wf->name) == 0 &&
20 (strcmp(wt->tab[i].hash, wf->hash) != 0 ||
21 wt->tab[i].mode != wf->mode)) {
22 return 1;
23 }
24 }
25 return 0;
26}

28// Note that append work_tree is safe about duplicates
29// if a file is in both work trees, it will be appended only once

31static void browse_and_append_wt_or_conflicts(work_tree_t *wt1,
32 work_tree_t *wt2,
33 list_t **conflicts,
34 work_tree_t *wt_without_conflicts)
35{
36 work_file_t *current_wf = NULL;
37
38 if (!wt1 || !wt2 || !conflicts || !wt_without_conflicts)
39 return;
40 for (size_t i = 0; i < wt1->size; ++i) {
41 current_wf = wt1->tab + i;
42 if (is_in_conflict(wt2, current_wf) &&
43 !search_data_in_list(*conflicts, current_wf->name)) {
44 create_and_insert_cell_in_list(*conflicts, current_wf->name);
45 }
46 else {
47 append_work_tree(wt_without_conflicts, current_wf->name, current_wf->hash,
48 current_wf->mode);
49 }
50 }
51}

53work_tree_t *mygit_merge_work_trees(work_tree_t *wt1, work_tree_t *wt2,
54 list_t **conflicts)
55{
56 work_tree_t *wt_without_conflicts = NULL;
57
58 if (!wt1 || !wt2 || !conflicts)
59 return NULL;
60 wt_without_conflicts = create_and_init_work_tree();
61 if (!wt_without_conflicts)
62 return NULL;
63 browse_and_append_wt_or_conflicts(wt1, wt2, conflicts, wt_without_conflicts);
64 browse_and_append_wt_or_conflicts(wt2, wt1, conflicts, wt_without_conflicts);
65 return wt_without_conflicts;
66}
