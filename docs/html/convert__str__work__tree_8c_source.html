84work_tree_t *convert_str_to_work_tree(const char *str)
85{
86 work_tree_t *wt = NULL;
87 char *str_cpy = NULL;
88 char *str_cpy_pointer = NULL;
89
90 if (!str)
91 return NULL;
92 str_cpy = strdup(str);
93 str_cpy_pointer = str_cpy;
94 if (!str_cpy)
95 return NULL;
96 wt = create_and_init_work_tree();
97 if (!wt) {
98 free(str_cpy);
99 return NULL;
100 }
101 if (!browse_string_and_append_wt(wt, &str_cpy)) {
102 free_work_tree(wt);
103 wt = NULL;
104 }
105 free(str_cpy_pointer);
