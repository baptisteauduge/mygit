34 add_handle_args,

41 checkout_branch_handle_args,
42 checkout_commit_handle_args,
43 merge_handle_args,
44 help_handle_args};
45 int nb_actions = 15;
46
47 for (int i = 0; i < nb_actions; ++i) {
48 if (strcmp(argv[1], actions_names[i]) == 0)
49 return actions_func[i](argc, argv);

51 fprintf(stderr, "Error: bad usage, `%s` is not an available action\n",
52 argv[1]);
