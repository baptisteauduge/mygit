73void restore_work_tree(work_tree_t *wt, const char *path)
74{
75 work_file_t *wf = NULL;
76 char *path_absolute_blob = NULL;
77 char *path_blob = NULL;
78
79 if (!wt || !path)
80 return;
81 for (size_t i = 0; i < wt->size; ++i) {
82 wf = &wt->tab[i];
83 path_blob = get_path_from_hash(wf->hash);
84 if (!path_blob)
85 return;
86 path_absolute_blob = get_path_absolute(path, path_blob);
87 free(path_blob);
88 if (!path_absolute_blob)
89 return;
90 if (is_save_file(path_absolute_blob))
91 set_content_file_mode_and_hash(wf, path_absolute_blob, path);
92 else
93 load_work_tree_and_iterate(wf, path_absolute_blob, path);
94 free(path_absolute_blob);
95 }
